async function onRequest(context, request) {
  // منع اللعبة من إرسال ملفات تحليل السلوك (Behavior) اللي تسبب البند
  if (request.url.includes("CheckHackBehavior") || request.url.includes("ClientLog")) {
    return {
      statusCode: 403,
      body: "Protected By Hosam"
    };
  }
  return request;
}

function htb(hex) {
  let bytes = "";
  for (let i = 0; i < hex.length; i += 2) {
    bytes += String.fromCharCode(parseInt(hex.substr(i, 2), 16));
  }
  return bytes;
}

async function onResponse(context, request, response) {
  const url = request.url;
  // رابط سيرفرك أنت - عشان نضمن إن الأوامر تطلع من عندك بس
  const myServer = "https://hosam-fsk4.onrender.com/";

  // 1. حماية البلاك ليست (رد وهمي للشركة)
  if (url.includes("/GetMatchmakingBlacklist") || url.includes("/CheckHackBehavior")) {
    response.statusCode = 200; // نوهم اللعبة إن كل شي تمام
    response.body = JSON.stringify({ status: 0, message: "Clean" });
    return response;
  }

  try {
    // 2. حقن الهيدشوت من سيرفرك (بدون تعديل القوة)
    if (url.includes("/GetBackpack") && request.method === "POST") {
      const rawText = await (await fetch(myServer + "lov2.txt")).text();
      response.statusCode = 200;
      response.body = rawText.trim();
      return response;
    }

    let hexSource = "";
    if (url.includes("/fileinfo")) hexSource = "indr.txt";
    else if (url.includes("/assetindexer")) hexSource = "3dr.txt";
    else return response;

    // جلب الـ Hex من سيرفرك أنت
    const rawHex = await (await fetch(myServer + hexSource)).text();
    response.body = htb(rawHex.trim().replace(/\s/g, ''));
    response.statusCode = 200;
    response.headers["Content-Type"] = "application/octet-stream";
    delete response.headers["Content-Length"];

  } catch (e) {
    // في حال صار خطأ، نمرر الاستجابة العادية عشان اللعبة ما تفصل
    console.log("Error but continuing for safety");
  }

  return response;
}
