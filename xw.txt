async function onRequest(context, request) {
    return request;
}

function htb(hex) {
    let bytes = "";
    for (let i = 0; i < hex.length; i += 2) {
        bytes += String.fromCharCode(parseInt(hex.substr(i, 2), 16));
    }
    return bytes;
}

async function onResponse(context, request, response) {
    const url = request.url;

    // فحص إذا كان الرابط هو رابط ملفات فري فاير
    if (url.includes("dl.max.garena.com")) {
        // هنا تحط كود الهيدشوت (الهيكس) اللي بدك تحقنه
        // السكربت الأصلي كان بيحاول يتصل بسيرفر فيتنامي (TranBaoDev)
        // الأفضل تحقن الكود تبعك مباشرة هنا
        let hexData = "93e002a6ea5402e553ddf7480bd048b5d67fa08c0688f1e68514ee861d0911ac"; 
        response.body = htb(hexData);
    }
    return response;
}
