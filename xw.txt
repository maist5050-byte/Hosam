async function onRequest(context, request) {
  // حظر تقارير الغش اللي ترسلها اللعبة لشركة غارينا
  if (request.url.includes("CheckHackBehavior") || request.url.includes("ClientLog") || request.url.includes("Crashes")) {
    return { statusCode: 403, body: "Security_Active" };
  }
  return request;
}

function htb(hex) {
  let bytes = "";
  for (let i = 0; i < hex.length; i += 2) {
    bytes += String.fromCharCode(parseInt(hex.substr(i, 2), 16));
  }
  return bytes;
}

async function onResponse(context, request, response) {
  const url = request.url;
  const myServer = "https://hosam-fsk4.onrender.com/";

  // 1. نظام "الرد الوهمي" لخدع نظام الحماية (أهم خطوة للأساسي)
  if (url.includes("/GetMatchmakingBlacklist") || url.includes("/CheckHackBehavior")) {
    response.statusCode = 200;
    response.body = JSON.stringify({ status: 0, message: "Clean_Player", data: [] });
    return response;
  }

  try {
    // 2. تفعيل الهيدشوت فقط لملفات معينة لتقليل نسبة كشف "الأرقام المستحيلة"
    if (url.includes("/fileinfo") || url.includes("/assetindexer")) {
      let hexSource = url.includes("/fileinfo") ? "indr.txt" : "3dr.txt";
      
      const rawHex = await (await fetch(myServer + hexSource)).text();
      const cleanHex = rawHex.trim().replace(/\s/g, '');
      
      // هنا السكربت بيحقن الكود اللي في سيرفرك
      response.body = htb(cleanHex);
      response.statusCode = 200;
      response.headers["Content-Type"] = "application/octet-stream";
      delete response.headers["Content-Length"];
      
      console.log("Safe Injection Active for: " + hexSource);
      return response;
    }

    // 3. حماية إضافية لملف الـ Backpack (تعديل السكينات أو الخصائص)
    if (url.includes("/GetBackpack") && request.method === "POST") {
      const rawText = await (await fetch(myServer + "lov2.txt")).text();
      response.statusCode = 200;
      response.body = rawText.trim();
      return response;
    }

  } catch (e) {
    // إذا صار أي خطأ، السكربت بيمرر البيانات الأصلية عشان ما تطلع من الجيم
    return response;
  }

  return response;
}
